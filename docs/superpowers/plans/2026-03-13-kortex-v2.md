# Kortex v2 Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Transform the Milowda/Antigravity markdown-based AI Agent Framework into a working web application with Kanban board, DAG pipeline engine, multi-model AI orchestration, and CLI agent protocol.

**Architecture:** Next.js 15 App Router full-stack app. SQLite + Drizzle ORM for persistence. Hybrid execution model — backend orchestrator for kickoff (document generation via AI APIs), CLI agents for development (code writing). SSE for real-time updates. Dark theme Linear-style UI.

**Tech Stack:** Next.js 15, React 19, Tailwind CSS v4, SQLite + better-sqlite3 + Drizzle ORM, @dnd-kit, Zod, nanoid, SSE, Geist font, Lucide icons, commander.js (CLI)

**Spec:** `docs/specs/2026-03-13-kortex-v2-design.md` — 1040 lines, 19 sections. READ THIS BEFORE IMPLEMENTING.

// src/lib/errors.ts

export class AppError extends Error {
  constructor(
    message: string,
    public statusCode: number = 500,
    public code: string = "INTERNAL_ERROR"
  ) {
    super(message);
    this.name = "AppError";
  }
}

export class NotFoundError extends AppError {
  constructor(resource: string, id: string) {
    super(`${resource} not found: ${id}`, 404, "NOT_FOUND");
    this.name = "NotFoundError";
  }
}

export class ValidationError extends AppError {
  constructor(message: string) {
    super(message, 400, "VALIDATION_ERROR");
    this.name = "ValidationError";
  }
}

export class TransitionError extends AppError {
  constructor(message: string) {
    super(message, 403, "TRANSITION_ERROR");
    this.name = "TransitionError";
  }
}

export class PermissionError extends AppError {
  constructor(message: string) {
    super(message, 403, "PERMISSION_ERROR");
    this.name = "PermissionError";
  }
}

export function errorResponse(error: unknown) {
  if (error instanceof AppError) {
    return Response.json(
      { error: { code: error.code, message: error.message } },
      { status: error.statusCode }
    );
  }
  console.error("Unhandled error:", error);
  return Response.json(
    { error: { code: "INTERNAL_ERROR", message: "Internal server error" } },
    { status: 500 }
  );
}

import { tasks } from "@/db/schema";
import { errorResponse } from "@/lib/errors";
import { createTaskSchema } from "@/lib/validators";
import { generateId } from "@/lib/id";

export async function GET(request: Request) {
  try {
    const url = new URL(request.url);
    const projectId = url.searchParams.get("projectId");
    const status = url.searchParams.get("status");
    const epicId = url.searchParams.get("epicId");

    const conditions = [];
    if (projectId) conditions.push(eq(tasks.projectId, projectId));
    if (status) conditions.push(eq(tasks.status, status));
    if (epicId) conditions.push(eq(tasks.epicId, epicId));

    const result =
      conditions.length > 0
        ? db
            .select()
            .from(tasks)
            .where(and(...conditions))
            .orderBy(tasks.orderIndex)
            .all()
        : db.select().from(tasks).orderBy(tasks.orderIndex).all();

    return Response.json({ data: result });
  } catch (error) {
    return errorResponse(error);
  }
}

export async function POST(request: Request) {
  try {
    const body = await request.json();
    const parsed = createTaskSchema.parse(body);

    const id = generateId();
    const testSteps = deriveTestSteps(parsed.labels);
    const now = new Date().toISOString();

    db.insert(tasks)
      .values({
        id,
        ...parsed,
        labels: JSON.stringify(parsed.labels),
        testSteps: JSON.stringify(testSteps),
        dependencies: JSON.stringify({ blocks: [], blockedBy: [], related: [] }),
        status: "backlog",
        createdAt: now,
        updatedAt: now,
      })
      .run();

    const created = db.select().from(tasks).where(eq(tasks.id, id)).get();
    return Response.json({ data: created }, { status: 201 });
  } catch (error) {
    return errorResponse(error);
  }
}

function deriveTestSteps(labels: string[]): string[] {
  const steps: string[] = [];
  if (labels.includes("needs-code-review")) steps.push("code_review");
  if (labels.includes("needs-qa")) steps.push("qa");
  if (labels.includes("needs-security")) steps.push("security");
  return steps;
}

### Task 21: Transition Engine

**Files:**
- Create: `src/lib/engine/transition.ts`
- Create: `src/app/api/v1/tasks/[id]/transition/route.ts`

Implements the state machine from spec §6.4. Validates persona permissions for each transition.

The transition rules:
```
backlog → todo                    : prime, project-manager
todo → in_progress                : assigned persona
in_progress → test_[first_step]   : assigned persona
in_progress → review              : assigned persona (no test steps)
test_X → test_Y                   : test persona (approved → next)
test_X → in_progress              : test persona (rejected → back)
test_[last_step] → review         : last test persona (approved)
review → done                     : prime or assigned reviewer
review → in_progress              : prime or reviewer (rejected)
any → backlog                     : prime, managers
```

- [ ] **Step 1: Create transition engine**

The engine should be a pure function: `validateTransition(task, toStatus, personaId) → {valid, reason?, newTestStep?}`. It checks:
1. Is the transition path valid?
2. Does the persona have permission?
3. For test transitions: what's the next test step?

- [ ] **Step 2: Create transition API endpoint**

POST `/api/v1/tasks/:id/transition` with `{toStatus, personaId}`. Uses the engine to validate, then updates the task.
